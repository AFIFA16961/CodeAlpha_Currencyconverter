import java. util.Scanner;

public class CurrencyConverter {

    public static void main(String[] args) {
        System.out.println(" **** Welcome to Currency Converter ****");
        System.out.println("***** Choose the currency type *****");
        System.out.println("1 Ruppe");
        System.out.println("2 Dollar");
        System.out.println("3 Euro");
        System.out.println("4 GBP");
        // Taking user inputs
        Scanner br = new Scanner(System.in);
        int ch = br.nextInt();
        System.out.println("Enter the amount");
        double Amt = br.nextDouble();
        // amount conversion
        switch (ch) {
            case 1:
                Ruppe(Amt);
                break;
            case 2:
                Dollar(Amt);
                break;
            case 3:
                Euro(Amt);
                break;
            case 4:
                GBP(Amt);
                break;
            default:
                System.out.println("Invalid choice");
        }

    }

    public static void Ruppe(double amt) {
        System.out.println("1 Ruppe = " + 0.012 + " Dollar");
        System.out.println();
        System.out.println(amt+" Ruppe = " + (amt*0.012) + " Dollar");
        System.out.println();

        System.out.println("1 Ruppe = " + 0.011 + " Euro");
        System.out.println();
        System.out.println(amt+" Ruppe = " + (amt*0.011) + " Euro");
        System.out.println();

        System.out.println("1 Ruppe = " + 0.0099 + " GBP");
        System.out.println();
        System.out.println(amt+" Ruppe = " + (amt*0.0099) + " GBP");
        System.out.println();

    }


    public static void Dollar(double amt) {
        System.out.println("1 Dollar = " + 83.26 + " Ruppe");
        System.out.println();
        System.out.println(amt+" Dollar = " + (amt*83.26) + " Ruppe");
        System.out.println();

        System.out.println("1 Dollar= " + 0.95 + " Euro");
        System.out.println();
        System.out.println(amt+" Dollar = " + (amt*0.95) + " Euro");
        System.out.println();

        System.out.println("1 Dollar = " + 0.82 + " GBP");
        System.out.println();
        System.out.println(amt+" Dollar = " + (amt*0.82) + " GBP");
        System.out.println();
    }



    public static void Euro(double amt){
        System.out.println("1 Euro = " + 88.06 + " Ruppe");
        System.out.println();
        System.out.println(amt+" Euro = " + (amt*88.06) + " Ruppe");
        System.out.println();

        System.out.println("1 Euro = " + 1.06 + " Dollar");
        System.out.println();
        System.out.println(amt+" Euro = " + (amt*1.06) + " Dollar");
        System.out.println();

        System.out.println("1 Euro = " + 0.87 + " GBP");
        System.out.println();
        System.out.println(amt+" Euro = " + (amt*0.87) + " GBP");
        System.out.println();
    }

    public static void GBP(double amt){
        System.out.println("1 GBP = " + 101.39 + " Ruppe");
        System.out.println();
        System.out.println(amt+" GBP = " + (amt*101.39) + " Ruppe");
        System.out.println();

        System.out.println("1 GBP = " + 1.22 + " Dollar");
        System.out.println();
        System.out.println(amt+" GBP = " + (amt*1.22) + " Dollar");
        System.out.println();

        System.out.println("1 GBP = " + 1.15 + " Euro");
        System.out.println();
        System.out.println(amt+" GBP = " + (amt*1.15) + " Euro");
        System.out.println();
    }
}
